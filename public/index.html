What is RightJS?

  • -It’s an alternative to jQuery
  • -It’s a DOM manipulation language
  • -It allows you to work with prototypes with more ease
  • -It is currently much faster than jQuery when doing things like selecting DOM elements
  • -It has UI extensions that are touch compatible
  • -Just include http://cdn.rightjs.org/right.js to get started!

            el : 'fbBlog',
            tagName: 'div',
            initialize : function(){
              this.fb_blog_temp = doT.template( $('templateBlog').html(), undefined);
            },
            render : function(){
              var that = this;
              _.each(this.model.models, function(fbPosts){
                that.el.append(that.fb_blog_temp(fbPosts.attributes));
              });
              return this;
            }
        });
